def creation_dataset(dataset, time_step=1):
    dataX = np.array([dataset[i:i+time_step, 0] for i in range(len(dataset)-time_step-1)])
    dataY = np.array([dataset[i+time_step, 0] for i in range(len(dataset)-time_step-1)])
    return dataX, dataY
