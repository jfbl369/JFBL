2.Data split: train/test by using TimeseriesSplit¶

In [5]:
#Split train_test via TimeSeriesSplit
from sklearn.model_selection import TimeSeriesSplit
import datetime as dt

tscv = TimeSeriesSplit(n_splits=5)
for train_index, test_index in tscv.split(df_LR):
    train_set = df_LR.iloc[train_index]
    test_set = df_LR.iloc[test_index].copy()
train=train_set.reset_index()
test=test_set.reset_index()

x_train = train["Date"].map(dt.datetime.toordinal)
y_train = train[["Close"]]
x_test = test["Date"].map(dt.datetime.toordinal)
y_test = test[["Close"]]
